# 数据结构刷题软件 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将《数据结构习题集》PDF（40页扫描图，9章511题）制作成单 HTML 文件 PWA 刷题 App，通过 GitHub Pages 部署。

**Architecture:** 单文件 `index.html` 内嵌全部 CSS/JS/数据/PWA manifest。封面页 → 答题页 → 结算页三屏切换，Canvas 动态生成 PWA 图标，localStorage 持久化进度。

**Tech Stack:** 纯 HTML5 + CSS3 + Vanilla JS（ES6+），零外部依赖，零 CDN，零框架。

---

## Global Constraints

- 单文件 `index.html`，所有资源内联（CSS/JS/图标/数据/manifest/SW）
- 零外部依赖，不引用任何 CDN 或 npm 包
- `display: standalone` PWA，Canvas 生成图标
- 暗色主题配色（背景 #1a1a2e，卡片 #16213e，正确 #00b894，错误 #e94560）
- 支持学习模式（即时反馈+解析）和考试模式（交卷后统一判分）
- 按章节刷题 + 随机抽题
- 所有题目附带 AI 生成解析
- 响应式适配：>768px PC 布局，≤768px 移动端布局，≤480px 紧凑布局

- [ ] **Step 5: 编写二叉树封面图标 Canvas 绘制代码**

```javascript
function drawTreeIcon() {
  const canvas = document.getElementById('treeIcon');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  const cw = canvas.width, ch = canvas.height;
  const cx = cw / 2, cy = 30;

  ctx.clearRect(0, 0, cw, ch);

  // 节点参数
  const nodeRadius = 14;
  const nodeColor = '#1a508b';
  const nodeStroke = '#6c5ce7';
  const lineColor = '#a0a0b0';
  const textColor = '#eaeaea';

  function drawNode(x, y, letter) {
    ctx.beginPath();
    ctx.arc(x, y, nodeRadius, 0, Math.PI * 2);
    ctx.fillStyle = nodeColor;
    ctx.fill();
    ctx.strokeStyle = nodeStroke;
    ctx.lineWidth = 2;
    ctx.stroke();
    ctx.fillStyle = textColor;
    ctx.font = 'bold 14px monospace';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(letter, x, y);
  }

  function drawLine(x1, y1, x2, y2) {
    ctx.beginPath();
    ctx.moveTo(x1, y1 + nodeRadius);
    ctx.lineTo(x2, y2 - nodeRadius);
    ctx.strokeStyle = lineColor;
    ctx.lineWidth = 2;
    ctx.stroke();
  }

  // 三层二叉树
  const level2y = cy + 40, level3y = cy + 80;
  drawNode(cx, cy, 'A');                         // 根
  drawLine(cx, cy, cx - 30, level2y);            // 连线
  drawLine(cx, cy, cx + 30, level2y);
  drawNode(cx - 30, level2y, 'B');               // 左子
  drawNode(cx + 30, level2y, 'C');               // 右子
  drawLine(cx - 30, level2y, cx - 50, level3y);  // 连线
  drawLine(cx - 30, level2y, cx - 10, level3y);
  drawLine(cx + 30, level2y, cx + 10, level3y);
  drawLine(cx + 30, level2y, cx + 50, level3y);
  drawNode(cx - 50, level3y, 'D');
  drawNode(cx - 10, level3y, 'E');
  drawNode(cx + 10, level3y, 'F');
  drawNode(cx + 50, level3y, 'G');
}

// 页面加载时绘制
document.addEventListener('DOMContentLoaded', drawTreeIcon);
```

```javascript
function generateAppIcon(size) {
  const canvas = document.createElement('canvas');
  canvas.width = size;
  canvas.height = size;
  const ctx = canvas.getContext('2d');

  // 背景：深蓝渐变圆角矩形
  const bgGrad = ctx.createLinearGradient(0, 0, size, size);
  bgGrad.addColorStop(0, '#1a1a2e');
  bgGrad.addColorStop(1, '#0f3460');
  ctx.fillStyle = bgGrad;
  ctx.beginPath();
  const r = size * 0.2;
  ctx.roundRect(0, 0, size, size, r);
  ctx.fill();

  // 绘制简化的二叉树白色图标
  const cx = size / 2, cy = size * 0.25;
  const nodeR = size * 0.08;
  const level2y = cy + size * 0.22;
  const level3y = cy + size * 0.44;

  function pNode(x, y) {
    ctx.beginPath();
    ctx.arc(x, y, nodeR, 0, Math.PI * 2);
    ctx.fillStyle = '#6c5ce7';
    ctx.fill();
    ctx.strokeStyle = '#eaeaea';
    ctx.lineWidth = Math.max(1, size * 0.005);
    ctx.stroke();
  }
  function pLine(x1, y1, x2, y2) {
    ctx.beginPath();
    ctx.moveTo(x1, y1 + nodeR);
    ctx.lineTo(x2, y2 - nodeR);
    ctx.strokeStyle = '#a0a0b0';
    ctx.lineWidth = Math.max(1, size * 0.004);
    ctx.stroke();
  }

  pNode(cx, cy);
  pLine(cx, cy, cx - size * 0.15, level2y);
  pLine(cx, cy, cx + size * 0.15, level2y);
  pNode(cx - size * 0.15, level2y);
  pNode(cx + size * 0.15, level2y);
  pLine(cx - size * 0.15, level2y, cx - size * 0.28, level3y);
  pLine(cx - size * 0.15, level2y, cx - size * 0.05, level3y);
  pLine(cx + size * 0.15, level2y, cx + size * 0.05, level3y);
  pLine(cx + size * 0.15, level2y, cx + size * 0.28, level3y);
  pNode(cx - size * 0.28, level3y);
  pNode(cx - size * 0.05, level3y);
  pNode(cx + size * 0.05, level3y);
  pNode(cx + size * 0.28, level3y);

  return canvas.toDataURL('image/png');
}
```

```javascript
function registerSW() {
  if (!('serviceWorker' in navigator)) return;

  const swCode = `
    const CACHE = 'ds-quiz-v1';
    self.addEventListener('install', e => {
      self.skipWaiting();
      e.waitUntil(caches.open(CACHE).then(c => c.addAll(['./'])));
    });
    self.addEventListener('activate', e => {
      e.waitUntil(caches.keys().then(keys =>
        Promise.all(keys.filter(k => k !== CACHE).map(k => caches.delete(k)))
      ));
      self.clients.claim();
    });
    self.addEventListener('fetch', e => {
      e.respondWith(
        caches.match(e.request).then(r => r || fetch(e.request).catch(() => r))
      );
    });
  `;

  const blob = new Blob([swCode], { type: 'application/javascript' });
  const swURL = URL.createObjectURL(blob);

  navigator.serviceWorker.register(swURL).catch(err => {
    console.warn('SW registration failed (expected on HTTP localhost):', err.message);
  });
}
```

```javascript
function filterQuestions(config) {
  let pool = [...ALL_QUESTIONS];

  // 按章节筛选
  if (config.chapter > 0) {
    pool = pool.filter(q => q.chapter === config.chapter);
  }

  // 随机打乱
  shuffleArray(pool);

  // 截取题数
  if (config.count > 0 && config.count < pool.length) {
    pool = pool.slice(0, config.count);
  }

  return pool;
}

function shuffleArray(arr) {
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
}
```

```javascript
function matchFillAnswer(userInput, correctAnswer) {
  const input = userInput.trim().replace(/\s+/g, ' ').toLowerCase();
```

```javascript
function showResult() {
  const total = state.results.length;
  const correct = state.results.filter(r => r.correct).length;
  const pct = total > 0 ? Math.round((correct / total) * 100) : 0;

  // 圆环动画
  const ring = document.getElementById('ringProgress');
  const circumference = 2 * Math.PI * 70; // r=70
  ring.style.strokeDasharray = circumference;
  ring.style.strokeDashoffset = circumference - (pct / 100) * circumference;
  document.getElementById('ringPercent').textContent = pct + '%';

  // 章节得分
  renderChapterScores();

  // 错题列表
  renderWrongReview();

  switchPage('quiz', 'result');
}

function renderChapterScores() {
  const byChapter = {};
  state.results.forEach(r => {
    const q = ALL_QUESTIONS.find(q => q.id === r.id);
    if (!q) return;
    if (!byChapter[q.chapter]) byChapter[q.chapter] = { total: 0, correct: 0, name: q.chapterName };
    byChapter[q.chapter].total++;
    if (r.correct) byChapter[q.chapter].correct++;
  });

  const html = Object.entries(byChapter).map(([ch, data]) => {
    const pct = Math.round((data.correct / data.total) * 100);
    return `
      <div class="chapter-score-row">
        <span class="chapter-score-name">${data.name}</span>
        <div class="chapter-score-bar">
          <div class="chapter-score-fill" style="width:${pct}%"></div>
        </div>
        <span class="chapter-score-pct">${pct}%</span>
      </div>
    `;
  }).join('');

  document.getElementById('chapterScores').innerHTML = html;
}

function renderWrongReview() {
  const wrongResults = state.results.filter(r => !r.correct);
  if (wrongResults.length === 0) {
    document.getElementById('wrongReview').innerHTML = `
      <div class="wrong-title">🎯 全部正确，太棒了！</div>
    `;
    return;
  }

  const items = wrongResults.map(r => {
    const q = ALL_QUESTIONS.find(q => q.id === r.id);
    if (!q) return '';
    return `
      <div class="wrong-item">
        <div class="question-text">${q.question}</div>
        <div class="wrong-detail">
          你的答案：<span>${r.userAnswer || '未作答'}</span> &nbsp;
          正确答案：<span class="correct-ans">${q.answer}</span>
        </div>
      </div>
    `;
  }).join('');

  document.getElementById('wrongReview').innerHTML = `
    <div class="wrong-title">📋 错题回顾 (${wrongResults.length})</div>
    ${items}
  `;
}
```

```javascript
const STORAGE_KEY = 'ds_quiz_data';

function saveProgress() {
  const data = {
    lastConfig: getConfig(),
    stats: loadStats(), // 读取现有统计再写回
    lastUpdated: Date.now(),
  };
  localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
}

function loadProgress() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : null;
  } catch {
    return null;
  }
}

function saveStats() {
  // 累加本次答题结果到历史统计
  const existing = loadStats();
  state.results.forEach(r => {
    if (!existing[r.id]) existing[r.id] = { correct: 0, wrong: 0 };
    if (r.correct) existing[r.id].correct++;
    else existing[r.id].wrong++;
  });

  const data = loadProgress() || {};
  data.stats = existing;
  data.lastConfig = getConfig();
  data.lastUpdated = Date.now();
  localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
}

function loadStats() {
  const data = loadProgress();
  return data?.stats || {};
}
```
